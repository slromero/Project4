# Define a function to replace non-values like 'N' or 'n' with an empty string using regex.fullmatch
def replace_with_null(value):
    if re.fullmatch(r'[/*]', str(value)):
        return ''
    elif re.fullmatch(r'[Nn]',str(value)):
        return ''
    else:
        return value
